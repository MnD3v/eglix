# Configuration de sécurité pour Eglix
# Ajoutez ces variables dans votre fichier .env ou dans votre dashboard Render

# Configuration HTTPS
APP_ENV=production
APP_DEBUG=false
APP_URL=https://eglix.lafia.tech

# Forcer HTTPS
FORCE_HTTPS=true

# Cookies sécurisés
SESSION_SECURE_COOKIE=true
SESSION_HTTP_ONLY=true
SESSION_SAME_SITE=lax

# Configuration des proxys (pour Render/Cloudflare)
TRUSTED_PROXIES=*
TRUSTED_HEADERS=X-Forwarded-For,X-Forwarded-Host,X-Forwarded-Port,X-Forwarded-Proto

# Configuration de la base de données
DB_CONNECTION=pgsql
DB_HOST=your-db-host
DB_PORT=5432
DB_DATABASE=your-db-name
DB_USERNAME=your-db-user
DB_PASSWORD=your-db-password

# Cache et sessions
CACHE_DRIVER=file
SESSION_DRIVER=file
SESSION_LIFETIME=120

# Mail (optionnel)
MAIL_MAILER=smtp
MAIL_HOST=your-smtp-host
MAIL_PORT=587
MAIL_USERNAME=your-email
MAIL_PASSWORD=your-password
MAIL_ENCRYPTION=tls
MAIL_FROM_ADDRESS=noreply@eglix.lafia.tech
MAIL_FROM_NAME="Eglix"

# Logs
LOG_CHANNEL=stack
LOG_LEVEL=error

# Instructions d'utilisation:
# 1. Copiez ces variables dans votre dashboard Render
# 2. Adaptez les valeurs selon votre configuration
# 3. Déployez votre application
# 4. Les en-têtes de sécurité seront automatiquement appliqués
